! ==================================================
! = YouTube
! ==================================================
! All rules in this section apply exclusively to YouTube.
! The goal is to reduce clutter, recommendations and
! unwanted content while keeping the platform functional.
! ==================================================


! --------------------------------------------------
! Live & Streamed Content
! --------------------------------------------------

! Hide live videos
www.youtube.com##:is(ytd-rich-item-renderer,ytd-grid-video-renderer,ytd-compact-video-renderer,ytd-item-section-renderer):has(.badge-style-type-live-now-alternate,.yt-spec-avatar-shape--live-ring)

! Hide VODs / streamed videos
www.youtube.com##:is(ytd-rich-item-renderer,ytd-grid-video-renderer,ytd-compact-video-renderer,ytd-item-section-renderer,ytd-video-renderer):has([aria-label~="Streamed"])

! Hide VODs / streamed videos (backup – less efficient)
www.youtube.com##:is(ytd-rich-item-renderer,ytd-grid-video-renderer,ytd-compact-video-renderer,ytd-item-section-renderer,ytd-video-renderer):not(:has([aria-label~="Streamed"])) :is(#metadata-line>span,.yt-content-metadata-view-model__metadata-row):last-of-type:has-text(Streamed):upward(ytd-rich-item-renderer,ytd-grid-video-renderer,ytd-compact-video-renderer,ytd-item-section-renderer,ytd-video-renderer)


! --------------------------------------------------
! Suggested & Recommended Videos
! --------------------------------------------------

! Subscriptions page – hide videos based on channel URLs
youtube.com##ytd-browse[page-subtype="subscriptions"] ytd-rich-item-renderer:has([href="/channel/channel1-ID-here"],[href="@/Channel2NameHere"])

! Subscriptions page – hide videos based on titles
youtube.com##ytd-browse[page-subtype="subscriptions"] ytd-rich-item-renderer:has(:is(#video-title-link,h3):is([title*="Partial Match"], [title*="Case-Insensitive Partial Match"i], [title~="Space-separated-aka-Whole-word-match"]))

! Sidebar – hide videos based on titles
youtube.com###related :is(ytd-compact-video-renderer,yt-lockup-view-model):has(:is(#video-title,h3):is([title*="Partial Match"], [title*="Case-Insensitive Partial Match"i], [title~="Space-separated-aka-Whole-word-match"]))

! Sidebar – hide videos based on channel names (procedural)
www.youtube.com###related :is(ytd-compact-video-renderer,yt-lockup-view-model)
  .yt-content-metadata-view-model__metadata-row:first-child>span
  :has-text(/^(Channel Name 1|Channel Name 2)$/)
  :upward(ytd-compact-video-renderer,yt-lockup-view-model)

! Watch page – hide sidebar
www.youtube.com##ytd-watch-flexy #secondary

! Watch page – sidebar without recommendations
www.youtube.com##ytd-watch-flexy #secondary


! --------------------------------------------------
! Sidebar Cleanup
! --------------------------------------------------

! Hide Explore section
youtube.com##ytd-guide-section-renderer:has(ytd-guide-entry-renderer a[title="Explore"])

! Hide "More from YouTube" section
youtube.com##ytd-guide-section-renderer:has(ytd-guide-entry-renderer a[title="YouTube Premium"])


! --------------------------------------------------
! Paid Promotion
! --------------------------------------------------

! Hide paid promotion overlay
youtube.com##.ytp-paid-content-overlay, .ytmPaidContentOverlayHost


! --------------------------------------------------
! Player UI & Endscreen Behavior
! --------------------------------------------------

! Hide endscreen recommendations
youtube.com##.html5-video-player.ended-mode .ytp-fullscreen-grid

! Fullscreen – remove recommendation grid when scrolling up
youtube.com##.html5-video-player.ytp-fullscreen:is(.ytp-fullscreen-grid-peeking,.ytp-fullscreen-grid-active) .ytp-fullscreen-grid
youtube.com##.html5-video-player.ytp-fullscreen:is(.ytp-fullscreen-grid-peeking,.ytp-fullscreen-grid-active) .ytp-gradient-bottom

! Fullscreen – keep controls visible and usable
youtube.com##.html5-video-player.ytp-fullscreen:is(.ytp-fullscreen-grid-peeking,.ytp-fullscreen-grid-active)
  .ytp-chrome-bottom:style(bottom: 0px !important;)

youtube.com##.html5-video-player.ytp-fullscreen:is(.ytp-fullscreen-grid-peeking,.ytp-fullscreen-grid-active)
  :not(:is(.ytp-autohide,.ytp-autohide-active))
  .ytp-chrome-bottom:style(opacity: unset !important;)

youtube.com##.html5-video-player.ytp-fullscreen:is(.ytp-fullscreen-grid-peeking,.ytp-fullscreen-grid-active)
  .ytp-overlays-container:style(bottom: calc(var(--yt-delhi-bottom-controls-height, 72px) + 30px) !important;)

youtube.com##.html5-video-player.ytp-fullscreen:is(.ytp-fullscreen-grid-peeking,.ytp-fullscreen-grid-active)
  :not(:is(.ytp-autohide,.ytp-autohide-active))
  .ytp-overlays-container:style(opacity: unset !important;)

! Disable grid scroll behavior
youtube.com##div.html5-video-player:style(--ytp-grid-scroll-percentage: 0 !important;)

! Ensure controls remain visible in fullscreen grid mode
youtube.com##div.html5-video-player.ytp-delhi-modern.ytp-fullscreen-grid-active
  .ytp-chrome-bottom:style(display: block !important;)

! WARNING:
! This filter disables the "wheel" event on the video player.
! If scrolling-related functionality breaks, remove this rule
! and rely on the cosmetic filters above instead.
www.youtube.com##+js(aeld, wheel, , elements, .html5-video-player )


! --------------------------------------------------
! Search Page Cleanup
! --------------------------------------------------

! Search – keep only videos (no shorts)
youtube.com##ytd-search ytd-item-section-renderer>#contents>:is(
  :not(ytd-video-renderer,yt-showing-results-for-renderer,[icon-name="promo-full-height:EMPTY_SEARCH"]),
  ytd-video-renderer:has([aria-label="Shorts"])
),ytd-secondary-search-container-renderer

! Search – hide results based on channel names
www.youtube.com##ytd-search ytd-video-renderer:has(
  #text.ytd-channel-name:is([title="Exact channel name 1"], [title="Exact channel name 2"])
)


! --------------------------------------------------
! Content Visibility & Layout
! --------------------------------------------------

! Hide members-only videos
youtube.com##ytd-rich-item-renderer:has(.yt-badge-shape--membership)

! Increase subscriptions grid to show 5 videos per row
www.youtube.com##ytd-browse[page-subtype="subscriptions"] ytd-rich-grid-renderer:style(--ytd-rich-grid-items-per-row: 5 !important;)
